#include <stdio.h>
#include <stdlib.h>

class arraySculptor {
private:
    char *arr;
    int symbols;
public:
    arraySculptor(int n) //конструктор без параметров
    {
        symbols = n;
        arr = new char[symbols];
    }

    arraySculptor(const arraySculptor& source) { //конструктор копирования
        symbols = source.symbols;
        arr = new char[symbols];
        for (int i = 0; i < symbols; i++)
            arr[i] = source.arr[i];
    }

    void sort() //сортировка массива по возрастанию
    {
        int temp; // переменные для хранения мин-ного значения и индекса массива
        for (int i = 0; i < symbols; i++)
        {
            temp = arr[i]; //инициализируем врем. переменную
            for (int j = i; j < symbols; j++)
                if (temp > arr[j])//находим минимальный элемент
                {
                    temp = arr[j]; //делаем перестановку
                    arr[j] = arr[i];
                    arr[i] = temp;
                }
        }
    }

    void print() {
        printf("Your array: ");
        for (int i = 0; i < symbols; i++)
            printf("%c", arr[i]);
        printf("\n");
    }

    char MAX(arraySculptor array) {
        arraySculptor tempArray(array);
        tempArray.sort();
        return tempArray.arr[symbols - 1];
    }

    /*char MIN(arraySculptor array) {

        tempArray.sort();
        return tempArray.arr[0];
    }*/
};

int main()

}
